    -- Fun√ß√µes auxiliares para encontrar a bola
    local function FindBall()
        local nearestBall = nil
        local minDist = math.huge
        
        for _, ball in pairs(Services.Workspace:GetChildren()) do
            if (ball.Name:lower():find("ball") or ball.Name:lower():find("soccer")) and ball:IsA("BasePart") then
                local dist = (HumanoidRootPart.Position - ball.Position).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestBall = ball
                end
            end
        end
        
        return nearestBall, minDist
    end
    
    local function FindGoal(ignoreOwn)
        local nearestGoal = nil
        local minDist = math.huge
        
        for _, obj in pairs(Services.Workspace:GetDescendants()) do
            local isGoal = obj.Name:lower():find("goal") or obj.Name:lower():find("gol")
            local isOwn = obj.Name:lower():find("own") or obj.Name:lower():find("own goal")
            
            if isGoal and (not ignoreOwn or not isOwn) and obj:IsA("BasePart") then
                local dist = (HumanoidRootPart.Position - obj.Position).Magnitude
                if dist < minDist then
                    minDist = dist
                    nearestGoal = obj
                end
            end
        end
        
        return nearestGoal
    end
